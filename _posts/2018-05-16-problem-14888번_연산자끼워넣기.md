# 14888번_연산자끼워넣기
### Algorithm Study(5.16)

[14888](https://www.acmicpc.net/problem/14888)

dfs와 백트래킹을 활용

```c++
#include <iostream>

using namespace std;

int N;
int num[101];
int cal[4];
//0 +
//1 -
//2 *
//3 /

int maxValue = -1000000000;;
int minValue = 1000000000;

void dfs(int result, int cnt) //0
{

	if (cnt == N-1) {
		if (maxValue < result)
			maxValue = result;
		if (minValue > result)
			minValue = result;
		return;
	}

	for (int i = 0; i < 4; i++)
	{
		if (cal[i] != 0) { //계산할 값이 남은경우
			cal[i]--; //연산자를 하나 사용
			if (i == 0)
				dfs(result + num[cnt + 1], cnt + 1);
			else if (i == 1)
				dfs(result - num[cnt + 1], cnt + 1);
			else if (i == 2)
				dfs(result*num[cnt + 1], cnt + 1);
			else if (i == 3)
				dfs(result / num[cnt + 1], cnt + 1);

			//복원
			cal[i] ++;
		}
	}
}

int main()
{
	cin >> N;
	for (int i = 0; i < N; i++)
	{
		cin >> num[i];
	}
	for (int i = 0; i < 4; i++)
	{
		cin >> cal[i];
	}

	dfs(num[0], 0);

	cout << maxValue << endl << minValue << endl;
}
```
