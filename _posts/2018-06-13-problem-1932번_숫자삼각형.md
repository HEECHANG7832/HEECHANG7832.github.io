[1932](https://www.acmicpc.net/problem/1932)


```c++
#include <iostream>
#include <algorithm>

using namespace std;

int main(void)
{
    int N;

    cin >> N;

    //input
    int **input;
    input = (int**)malloc(sizeof(int*)*N);
    for (int i = 0; i < N; i++)
    {
        input[i] = (int*)malloc(sizeof(int)*(N+2)); //양쪽에 0을 추가하기 위해서
    }

    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < i+3; j++)
        {
            if (j == 0)
            {
                input[i][j] = 0;
                continue;
            }
            if (j == i+2)
            {
                input[i][j] = 0;
                continue;
            }
            cin >> input[i][j];
        }
    }


    //DP
    int maxValue = 0;
    for (int i = 1; i < N; i++)
    {
        for (int j = 0; j < i+3; j++)
        {
            int value = max(input[i - 1][j - 1], input[i - 1][j]);
            input[i][j] += value;
            if (input[i][j] > maxValue)
                maxValue = input[i][j];
        }
    }

    /*for (int i = 0; i < N; i++)
    {
    for (int j = 0; j < i+3; j++)
    {
    cout << input[i][j] << " ";
    }
    cout << endl;
    }*/

    cout << maxValue << endl;

}
```
