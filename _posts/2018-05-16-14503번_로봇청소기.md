# 14503번_로봇청소기
### Algorithm Study(5.16)

[14503](https://www.acmicpc.net/problem/14503)

```c++
#include <iostream>

using namespace std;

int map[51][51];
int clean[51][51];

int N, M;
int sx, sy, sd;

int mapcount = 1;

int dy[] = { -1, 0, 1, 0 };
int dx[] = { 0, 1, 0, -1 };

void robo(int y, int x, int d)
{
	if (clean[y][x] == 0)
		clean[y][x] = mapcount++; //청소

	int dir = d;
	for (int k = 0; k < 4; k++)
	{
		dir = (dir + 3) % 4;

		int ly = y + dy[dir];
		int lx = x + dx[dir];

		//왼쪽에 청소할게 있으면
		if (ly > 0 && lx > 0 && ly < N - 1 && lx < M - 1 && map[ly][lx] != 1) {
			if (clean[ly][lx] == 0) {
				robo(ly, lx, dir);
				return;
			}
		}
	}

	int back = (d + 2) % 4;
	int by = y + dy[back];
	int bx = x + dx[back];
	if (map[by][bx] == 1) {
		//cout << "finish" << endl;
		return;
	}
	else {
		robo(by, bx, d);
	}
}

int main()
{

	cin >> N >> M;

	cin >> sy >> sx >> sd;

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < M; j++)
		{
			cin >> map[i][j];
			clean[i][j] = map[i][j];
		}
	}

	robo(sy, sx, sd);

	/*for (int i = 0; i < N; i++)
	{
	for (int j = 0; j < M; j++)
	{
	cout << clean[i][j] << " ";
	}cout << endl;
	}*/
	cout << mapcount - 1 << endl;
}
```
