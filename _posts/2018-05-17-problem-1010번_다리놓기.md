[1010](https://www.acmicpc.net/problem/1010)

Coose연산을 할때 팩토리얼 연산이 있어서 오버플로우가 발생할 수 있다.
이대 팩토리얼 연산을 피해가는 방법

**반복문 내부동작**
result = 1;
result *= 8;
result /= 1;
result *= 7;
result /= 2;
result *= 6;
result /= 3;
result *= 5;
result /= 4;


```c++
#include <iostream>
#include <algorithm>

using namespace std;

long long int Comb(int n, int r) {

	long long int k = 1;

	if (r > n) return 0;
	for (int i = 1; i <= r; i++)
	{
		k *= n--;
		k /= i;
	}
	return k;
};

int main()
{
	int testcase;
	cin >> testcase;
	for (int i = 0; i < testcase; i++)
	{
		int r, n;
		cin >> r >> n;
		cout << Comb(n,r) << endl;
	}
}
```
