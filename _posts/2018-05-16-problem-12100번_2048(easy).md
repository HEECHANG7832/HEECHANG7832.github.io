```c++

#include <iostream>
#include <algorithm>

#define UP 0
#define DOWN 1
#define LEFT 2
#define RIGHT 3

//Tip.
//int arr[][] 대신 vector<vector<int>> 사용하면
//call by value로 들어간다

using namespace std;

int N;

int moveLimit = 5;
int ans = 0;

int findMax(int(*arr)[22])
{
	int isMax = 0;
	for (int j = 0; j < N; j++)
	{
		for (int k = 0; k < N; k++)
		{
			if (isMax < arr[j][k]) {
				isMax = arr[j][k];
			}
		}
	}
	return isMax;
}
void copyArr(int(*operand)[22], int(*src)[22])
{
	for (int j = 0; j < N; j++)
	{
		for (int k = 0; k < N; k++)
		{
			operand[j][k] = src[j][k];
		}
	}
}

void dfs(int d, int(*arr)[22], int num)
{
	switch (d)
	{
	case UP:
		//0과 1이 같으면 합친다.
		for (int k = 0; k < N; k++)
		{
			for (int i = 0; i < N - 1; i++)
			{
				//나머지를 올린다.
				for (int j = i + 1; j < N; j++)
				{
					if (arr[j][k] != 0) {
						int t = j;
						while (1) {
							if (t <= 0)
								break;
							if (arr[t - 1][k] == 0) {
								arr[t - 1][k] = arr[t][k];
								arr[t][k] = 0;
							}
							else { //윗칸이 0이 아니면 올리기 중지
								break;
							}t--;
						}
					}
				}
				//같으면 합치고
				if (arr[i][k] == arr[i + 1][k]) {
					arr[i][k] *= 2;
					arr[i + 1][k] = 0;
				}
			}
		}
		break;
	case DOWN:
		for (int k = 0; k < N; k++)
		{
			for (int i = N - 1; i > 0; i--)
			{
				for (int j = i - 1; j >= 0; j--)
				{
					if (arr[j][k] != 0) {
						int t = j;
						while (1) {
							if (t >= N - 1)
								break;
							if (arr[t + 1][k] == 0) {
								arr[t + 1][k] = arr[t][k];
								arr[t][k] = 0;
							}
							else {
								break;
							}t++;
						}
					}
				}
				if (arr[i][k] == arr[i - 1][k]) {
					arr[i][k] *= 2;
					arr[i - 1][k] = 0;
				}
			}
		}
		break;
	case LEFT:
		for (int k = 0; k < N; k++)
		{
			for (int i = 0; i < N - 1; i++)
			{
				for (int j = i + 1; j < N; j++)
				{
					if (arr[k][j] != 0) {
						int t = j;
						while (1) {
							if (t <= 0)
								break;
							if (arr[k][t - 1] == 0) {
								arr[k][t - 1] = arr[k][t];
								arr[k][t] = 0;
							}
							else {
								break;
							}t--;
						}
					}
				}
				if (arr[k][i] == arr[k][i + 1]) {
					arr[k][i] *= 2;
					arr[k][i + 1] = 0;
				}
			}
		}
		break;
	case RIGHT:
		for (int k = 0; k < N; k++)
		{
			for (int i = N - 1; i > 0; i--)
			{
				for (int j = i - 1; j >= 0; j--)
				{
					if (arr[k][j] != 0) {
						int t = j;
						while (1) {
							if (t >= N - 1)
								break;
							if (arr[k][t + 1] == 0) {
								arr[k][t + 1] = arr[k][t];
								arr[k][t] = 0;
							}
							else {
								break;
							}t++;
						}
					}
				}
				if (arr[k][i] == arr[k][i - 1]) {
					arr[k][i] *= 2;
					arr[k][i - 1] = 0;
				}
			}
		}
		break;
	default:
		break;
	}

	if (num >= moveLimit) { //5번째 이동인경우
		ans = max(ans, findMax(arr)); //최대값찾아서 반환
		return;
	}

	/*for (int i = 0; i < N; i++)
	{
	for (int j = 0; j < N; j++)
	{
	cout << arr[i][j] << " ";
	}cout << endl;
	}cout << endl;*/

	int temp[22][22] = { 0, };
	//상 하 좌 우
	for (int i = 0; i < 4; i++)
	{
		copyArr(temp, arr);
		dfs(i, temp, num + 1);
	}

}

int main()
{
	int input[22][22] = { 0, };

	cin >> N;

	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			cin >> input[i][j];
		}
	}

	dfs(-1, input, 0);

	cout << ans << endl;
}
```
