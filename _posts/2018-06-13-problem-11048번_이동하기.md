```c++
#include <iostream>

using namespace std;

int max(int a, int b, int c)
{
    if (a >= b && a >= c)
        return a;
    else if (b >= a && b >= c)
        return b;
    else if (c >= a && c >= b)
        return c;
}
int main(void)
{
    int N, M;
    int **input;

    cin >> N;
    cin >> M;

    input = (int**)malloc(sizeof(int*)*N + 1);
    for (int i = 0; i <= N; i++)
        input[i] = (int*)malloc(sizeof(int)*M + 1);

    for (int i = 0; i <= N; i++)
        input[i][0] = 0;
    for (int i = 0; i <= M; i++)
        input[0][i] = 0;

    for (int i = 1; i <= N; i++)
    {
        for (int j = 1; j <= M; j++)
        {
            cin >> input[i][j];
        }
    }

    /*for (int i = 0; i <= N; i++)
    {
    for (int j = 0; j <= M; j++)
    {
    cout << input[i][j];
    }
    cout << endl;
    }*/

    for (int i = 1; i <= N; i++)
    {
        for (int j = 1; j <= M; j++)
        {
            input[i][j] = max(input[i - 1][j - 1] + input[i][j], input[i - 1][j] + input[i][j], input[i][j - 1] + input[i][j]);
        }
    }

    cout << input[N][M] << endl;
    return 0;
}
```
