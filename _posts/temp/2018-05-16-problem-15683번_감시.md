[15683](https://www.acmicpc.net/problem/15683)

```c++
#include <iostream>
#include <vector>
#include <utility>
#include <string.h>

using namespace std;

int map[9][9];
int copyMap[9][9];

vector<pair<int, int>> cctv;
vector<int> direction;
int n, m;

int dx[] = {1, 0, -1, 0};
int dy[] = {0, 1, 0, -1};

int minValue = 987654321;

//d방향으로 한줄 그린다
void drawLine(int x, int y, int d)
{
	//현재 방향으로 계속 #삽입
	int nx = x + dx[d];
	int ny = y + dy[d];
	while (1) {
		//지도를 벗어나지 않고
		if (nx >= 0 && ny >= 0 && nx < 9 && ny < 8) {
			//벽을 만나면 그만
			if (copyMap[nx][ny] == 6) {
				break;
			}
			//cctv가 아니면
			if (copyMap[nx][ny] == 0) {
				copyMap[nx][ny] = 7; //감시영역
			}
			//한칸더 진행
			nx += dx[d];
			ny += dy[d];		
		}
		//지도 밖이면
		else {
			break;
		}
	}
}
void drawMap()
{
	//원본 보존
	memcpy(copyMap, map, sizeof(map));

	for (int i = 0; i < cctv.size(); i++)
	{
		pair<int, int> p = cctv.at(i);
		int d = direction.at(i);

		int x = p.first;
		int y = p.second;

		//cctv종류에 따라서
		switch (copyMap[x][y])
		{
		case 1:
			drawLine(x, y, d);
			break;
		case 2:
			drawLine(x, y, d);
			drawLine(x, y, (d+2)%4);
			break;
		case 3:
			drawLine(x, y, d);
			drawLine(x, y, (d+1)%4);
			break;
		case 4:
			drawLine(x, y, d);
			drawLine(x, y, (d+1)%4);
			drawLine(x, y, (d+2)%4);
			break;
		case 5:
			drawLine(x, y, d);
			drawLine(x, y, (d + 1) % 4);
			drawLine(x, y, (d + 2) % 4);
			drawLine(x, y, (d + 3) % 4);
			break;
		default:
			break;
		}
	}	//그림을 다 그림


	//출력
	int count = 0;
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			//cout << copyMap[i][j] << " ";
			if (copyMap[i][j] == 0) {
				count++;
			}
		}//cout << endl;
	}//cout << endl;

	if (count < minValue)
		minValue = count;
}

void countEmptyPlace(int index)
{
	if (index == cctv.size()) {
		drawMap();
		return;
	}

	pair<int, int> position = cctv.at(index);
	int kindOfCCTV = map[position.first][position.second];

	//cctv종류에 따라서
	switch (kindOfCCTV)
	{
	case 1:
		for (int i = 0; i < 4; i++)
		{
			direction.push_back(i);
			countEmptyPlace(index + 1);
			direction.pop_back();
		}
		break;
	case 2:
		for (int i = 0; i < 2; i++)
		{
			direction.push_back(i);
			countEmptyPlace(index + 1);
			direction.pop_back();
		}
		break;
	case 3:
		for (int i = 0; i < 4; i++)
		{
			direction.push_back(i);
			countEmptyPlace(index + 1);
			direction.pop_back();
		}
		break;
	case 4:
		for (int i = 0; i < 4; i++)
		{
			direction.push_back(i);
			countEmptyPlace(index + 1);
			direction.pop_back();
		}
		break;
	case 5:
		direction.push_back(0);
		countEmptyPlace(index + 1);
		direction.pop_back();
		break;

	default:
		break;
	}
}

int main()
{
	cin >> n >> m;

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			cin >> map[i][j];
			//cctv 위치
			if (map[i][j] != 0 && map[i][j] != 6) {
				cctv.push_back(make_pair(i, j));
			}
		}
	}

	countEmptyPlace(0);

	cout << minValue << endl;
}
```
